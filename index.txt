    <script src="https://cdnjs.cloudflare.com/ajax/libs/jquery/3.2.1/jquery.min.js"></script>
    <script type="text/javascript">


        var topics = ["dogs", "horses", "elephants", "rabbits", "lions", "monkeys"];


        function displayNewGif() {
            var animals = $(this).attr("data-name");

            var queryURL = "https://api.giphy.com/v1/gifs/search?q=" + animals + "&api_key=6j59bfFQh1ukbUtOetli3m2wX18atuGV&limit=10";



            //AJAX CALL 
            $.ajax({
                url: queryURL,
                method: "GET"
            }).then(function (response) {
                //console.log(response);

                var results = response.data;

                //looping through the produced images
                for (var i = 0; i < results.length; i++) {
                    //create a div for the gifs
                    var animalDiv = $("<div class='animal'>");

                    //get rating info
                    var rating = response.data[i].rating;

                    var p1 = $("<p>").text("Rating: " + rating);

                    //get images
                    var animatedGif = response.data[i].images.original.url;

                    var stillGif = response.data[i].images.original_still.url;

                    var animalImage = $("<img>").attr("src", stillGif);

                    animalImage.attr("data-still", stillGif);

                    animalDiv.attr("data-animate", animatedGif);

                    //display image and rating
                    animalDiv.append(p1, animalImage);

                    $("#giphy-imgs").prepend(animalDiv);



                }
            })
        }
        function renderButtons() {

            $("#topics-buttons").empty();

            for (var i = 0; i < topics.length; i++) {

                var topicsButtons = $("<button>");

                topicsButtons.addClass("inputButtons");

                topicsButtons.attr("data-name", topics[i]);

                topicsButtons.text(topics[i]);

                $("#topics-buttons").append(topicsButtons);
            }
        }

        $("#add-animal").on("click", function (event) {

            event.preventDefault();

            var animals = $("#topics-input").val().trim();

            topics.push(animals);

            renderButtons();

        })


        //click to play and stop
        $(".animal").on("click", function () {  // <----NEED HELP

            var state = $(this).attr("data-state");

            if (state === "still") {

                $(this).attr("src", $(this).attr("data-animate"));

                $(this).attr("data-state", "animate");

            } else {

                $(this).attr("src", $(this).attr("data-still"));

                $(this).attr("data-state", "still");
            }
        });

        $(document).on("click", ".inputButtons", displayNewGif);

        renderButtons();


    </script>
